Step 2 — Screen layout constants
Add this FixedPool to UI.ailang near the top, after existing pools.
Values are computed from a default 220×50 terminal; the SIGWINCH handler
must recompute them at runtime (see Step 3).
FixedPool.UILayout {
    // Populated at runtime by UI_ComputeLayout().
    // Defaults assume 220 cols x 50 rows.
    "sidebar_w":   Initialize=24
    "chat_x":      Initialize=25   // sidebar_w + 1 (border col)
    "chat_w":      Initialize=195  // cols - sidebar_w - 1
    "prompt_rows": Initialize=4
    "chat_rows":   Initialize=46   // rows - prompt_rows
    "border_col":  Initialize=25   // the vertical divider column
}

Step 3 — UI_ComputeLayout
Add this function to UI.ailang. Call it from UI.Init() after
TUI_GetScreenSize(), and again inside the SIGWINCH handler.
Function.UI_ComputeLayout {
    Body: {
        cols = TUI_GetWidth()
        rows = TUI_GetHeight()

        sidebar_w   = 24
        prompt_rows = 4
        chat_rows   = Subtract(rows, prompt_rows)
        border_col  = Add(sidebar_w, 1)
        chat_x      = Add(sidebar_w, 2)
        chat_w      = Subtract(cols, Add(sidebar_w, 1))

        UILayout.sidebar_w   = sidebar_w
        UILayout.chat_x      = chat_x
        UILayout.chat_w      = chat_w
        UILayout.prompt_rows = prompt_rows
        UILayout.chat_rows   = chat_rows
        UILayout.border_col  = border_col

        // Register panes with Layout library
        TUI_Layout_SetPane(TUI_Pane.SIDEBAR, 1,          1,         sidebar_w, chat_rows)
        TUI_Layout_SetPane(TUI_Pane.CHAT,    chat_x,     1,         chat_w,    chat_rows)
        TUI_Layout_SetPane(TUI_Pane.PROMPT,  1,          Add(chat_rows, 1), cols, prompt_rows)

        // Chat scroll region excludes the prompt rows
        TUI_Layout_SetScroll(TUI_Pane.CHAT, 1, chat_rows)

        ReturnValue(0)
    }
}

Step 5 — Draw the chrome
Add function UI_DrawChrome to UI.ailang. Call it once from UI.Init()
after layout is computed, and again after every SIGWINCH.
Function.UI_DrawChrome {
    Body: {
        cols      = TUI_GetWidth()
        rows      = TUI_GetHeight()
        chat_rows = UILayout.chat_rows
        border    = UILayout.border_col

        // Vertical divider between sidebar and chat
        TUI_Color_Border()
        TUI_Draw_VLine(1, border, chat_rows)

        // Sidebar header: "  FILES" dimmed
        TUI_Color_SidebarDim()
        TUI_Draw_PrintClipped(1, 1, "  FILES", UILayout.sidebar_w)

        // Divider under sidebar header
        TUI_Color_Border()
        TUI_Draw_HLine(2, 1, Subtract(UILayout.sidebar_w, 1))

        // Bottom rule separating chat from prompt
        TUI_Draw_HLine(Add(chat_rows, 1), 1, cols)

        TUI_ResetColor()
        TUI_Flush()
        ReturnValue(0)
    }
}

Step 7 — Wire up the file tree
In IPCDispatch.ailang, find the point where a tool result is received
(after the IPC read returns the result envelope). Add a call to mark the
file dirty whenever the tool is Write or Edit:
// After result received, before returning to caller:
IfCondition Or(EqualTo(tool_name, "Write"), EqualTo(tool_name, "Edit")) ThenBlock: {
    // 'path' should be the args.path string from the call envelope
    TUI_Tree_MarkDirty(call_path)
    TUI_SaveCursor()
    TUI_Tree_Render(TUI_Pane.SIDEBAR)
    TUI_Flush()
    TUI_RestoreCursor()
}
Where call_path is the path argument extracted from the tool call
args. Read the existing args parsing in IPCDispatch.ailang to find the
correct variable name — do not guess.
Also populate the tree on startup. In ClaudeCode.ailang, after IPC
services are launched, add an LS call to the project root and feed the
result into TUI_Tree_Add. Walk one level deep only for the initial
population (directories + files at root). Set FL_DIR flag for
directories, 0 for files. depth=0 for all initial entries.

Step 8 — Wire up tool blocks
In IPCDispatch.ailang:
Before dispatching a tool call, add:
block_idx = TUI_Block_Begin(tool_name)
After the IPC result is received, add:
TUI_Block_End(block_idx, result_ok, elapsed_ms)
TUI_Block_AppendOutput(block_idx, result_content)
Where:

result_ok is 1 if ok:true in the result envelope, 0 otherwise
elapsed_ms is wall-clock ms between dispatch and result (use
clock_gettime(CLOCK_MONOTONIC) via syscall 228 if not already
measured; store seconds×1000 + nanos/1000000)
result_content is the content string from the result envelope

Then render the blocks into the chat area above the prompt. Add a
function UI_RenderToolBlocks to UI.ailang:
Function.UI_RenderToolBlocks {
    Body: {
        // Render in the bottom N rows of the chat pane,
        // above the horizontal rule.
        // max_rows = 8 to avoid crowding the chat stream.
        start = Subtract(UILayout.chat_rows, 9)
        TUI_Block_Render(start, UILayout.chat_w, 8)
        TUI_Flush()
        ReturnValue(0)
    }
}
Call UI_RenderToolBlocks once after TUI_Block_End in the dispatch
path, wrapped in TUI_SaveCursor / TUI_RestoreCursor.
